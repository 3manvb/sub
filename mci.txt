# تنظیم نام کاربری و نام مخزن گیت‌هاب
GITHUB_USERNAME="3manvb"
GITHUB_REPO="mci.txt"
GITHUB_TOKEN="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"

# آدرس فایلی که می‌خواهید دانلود کنید
FILE_URL="https://3manvb.s3.ir-thr-at1.arvanstorage.ir/mci.txt"

# نام فایل پس از دانلود
FILE_NAME="mci.txt"

# دانلود فایل
curl -L $FILE_URL -o $FILE_NAME

# اضافه کردن فایل به مخزن گیت‌هاب
git add $FILE_NAME
git commit -m "افزودن فایل جدید"
git push

# API URL برای ایجاد فایل جدید در مخزن گیت‌هاب
API_URL="https://api.github.com/repos/$GITHUB_USERNAME/$GITHUB_REPO/contents/$FILE_NAME"

# ایجاد فایل جدید در مخزن گیت‌هاب با استفاده از API
curl -X PUT -H "Authorization: token $GITHUB_TOKEN" -d "{\"path\": \"$FILE_NAME\", \"message\": \"ایجاد فایل جدید\", \"content\": \"$(base64 $FILE_NAME)\"}" $API_URL
